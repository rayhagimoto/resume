\BLOCK{import 'partials/macros.tex' as macros}

% Redefine bibliography label to use "1." instead of "[1]"
\makeatletter
\renewcommand\@biblabel[1]{#1.\hfill}
\makeatother

% Redefine thebibliography environment to match enumerate style
\makeatletter
\renewenvironment{thebibliography}[1]
  {\begin{list}{\@biblabel{\arabic{enumi}}}%
    {\setlength{\labelwidth}{2em}%
     \setlength{\leftmargin}{2.5em}%
     \setlength{\labelsep}{0.5em}%
     \setlength{\itemsep}{0.6em}%
     \setlength{\parsep}{0pt}%
     \usecounter{enumi}}%
     \sloppy\clubpenalty4000\widowpenalty4000\sfcode`\.\@m}
  {\end{list}}
\makeatother
