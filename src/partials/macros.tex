\BLOCK{macro BaseSection(title)}
  \begin{TightFrame}
    \section{\VAR{title}}
    \VAR{caller()}
  \end{TightFrame}
\BLOCK{endmacro}

\BLOCK{macro BaseEntry(title, dates, subtitle, location, body)}%
  \begin{TightFrame}%
    {%
    \BoldBeforeComma{\VAR{title}}%
    \hfill
    \VAR{dates}\\%
    \textit{\VAR{subtitle}\BLOCK{if location}, \VAR{location}\BLOCK{endif}}%
    \begin{TightFrame}%
      {%
        \VAR{body}%
      }%
    \end{TightFrame}%
    }%
  \end{TightFrame}%
\BLOCK{endmacro}%

\BLOCK{macro AddBullets(items)}
  \begin{itemize}[leftmargin=1em]
    \BLOCK{for item in items}
      \item \VAR{item}
    \BLOCK{endfor}
  \end{itemize}
\BLOCK{endmacro}

\BLOCK{macro AddDescription(text)}
  \VAR{text}
  \vspace{0.5em}
\BLOCK{endmacro}
