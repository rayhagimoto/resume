\BLOCK{macro BaseSection(title)}
  \begin{TightFrame}
    \section{\VAR{title}}
    \VAR{caller()}
  \end{TightFrame}
\BLOCK{endmacro}

\BLOCK{macro BaseEntry(title, dates, subtitle, location, body)}
  \begin{TightFrame}
    \makebox[0pt][l]{%
      \parbox{\linewidth}{%
        \BoldBeforeComma{\VAR{title}}%
        \hfill
        \VAR{dates}\\%
        \textit{\VAR{subtitle}\BLOCK{if location}, \VAR{location}\BLOCK{endif}}%
        \begin{TightFrame}
          \VAR{body}
        \end{TightFrame}
      }%
    }%
  \end{TightFrame}
\BLOCK{endmacro}

\BLOCK{macro AddBullets(items)}
  \begin{itemize}[leftmargin=1em]
    \BLOCK{for item in items}
      \item \VAR{item}
    \BLOCK{endfor}
  \end{itemize}
\BLOCK{endmacro}

\BLOCK{macro AddDescription(text)}
  \VAR{text}
  \vspace{0.5em}
\BLOCK{endmacro}

\BLOCK{macro SkillsEntry(skills)}
  \noindent{\textbf{Skills:}}
  \BLOCK{for skill in skills}
    \VAR{skill}\BLOCK{if not loop.last}, \BLOCK{endif}
  \BLOCK{endfor}
\BLOCK{endmacro}

\BLOCK{macro AwardsEntry(awards)}
  \begin{itemize}
    \BLOCK{for award in awards}
      \item \VAR{award}
    \BLOCK{endfor}
  \end{itemize}
\BLOCK{endmacro}
